## Install

```bash
npm install --save typescript-packate2
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'typescript-packate2'
import 'typescript-packate2/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
